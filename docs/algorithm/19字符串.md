## leetCode 151  翻转字符串里的单词
第一版
```js
var reverseWords = function(s) {
    let strArr = s.split(' ')
    let res = []
    for(let i = strArr.length - 1; i >=0 ; i--) {
        if(strArr[i] !== ' ' && strArr[i] !== '') {
            res.push(strArr[i])
        }
    }
   return res.join(' ')
}
```
第二版: 双指针
```js
 var reverseWords = function(s) {
   // 字符串转数组
   const strArr = Array.from(s);
   // 移除多余空格
   removeExtraSpaces(strArr);
   // 翻转
   reverseWords(strArr, 0, strArr.length - 1);

   let start = 0;

   for(let i = 0; i <= strArr.length; i++) {
     if (strArr[i] === ' ' || i === strArr.length) {
       // 翻转单词
       reverseWords(strArr, start, i - 1);
       start = i + 1;
     }
   }

   return strArr.join('');
};

// 删除多余空格
function removeExtraSpaces(strArr) {
  let slowIndex = 0;
  let fastIndex = 0;

  while(fastIndex < strArr.length) {
    // 移除开始位置和重复的空格
    if (strArr[fastIndex] === ' ' && (fastIndex === 0 || strArr[fastIndex - 1] === ' ')) {
      fastIndex++;
    } else {
      strArr[slowIndex++] = strArr[fastIndex++];
    }
  }

  // 移除末尾空格
  strArr.length = strArr[slowIndex - 1] === ' ' ? slowIndex - 1 : slowIndex;
}

// 翻转从 start 到 end 的字符
function reverseWords(strArr, start, end) {
  let left = start;
  let right = end;

  while(left < right) {
    // 交换
    [strArr[left], strArr[right]] = [strArr[right], strArr[left]];
    left++;
    right--;
  }
}
```

## leetCode 389 找不同
方法一: 计数
```js
var findTheDifference = function(s, t) {
  const cnt = new Array(26).fill(0)
  for(const ch of s) {
    cnt[ch.charCodeAt() - 'a'.charCodeAt()]++
  }
  for(const ch of t) {
    cnt[ch.charCodeAt() - 'a'.charCodeAt()]--
    if(cnt[ch.charCodeAt() - 'a'.charCodeAt()]-- < 0) {
      return ch
    }
  }
  return ' '
}
```
方法二: 求和
```js
var findTheDifference = function(s, t) {
  let as = 0, at = 0
  for(let i = 0; i < s.length; i++) {
    as += s[i].charCodeAt()
  }
  for(let i = 0; i < t.length; i++) {
    at += t[i].charCodeAt()
  }
  return String.formCharCode(at - as)
}
```
## leetCode 953 验证外星语词典
```js
var isAlienSorted = function(words, order) {
    const index = new Array(26).fill(0);
    for (let i = 0; i < order.length; ++i) {
        index[order[i].charCodeAt() - 'a'.charCodeAt()] = i;
    }
    for (let i = 1; i < words.length; i++) {
        let valid = false;
        for (let j = 0; j < words[i - 1].length && j < words[i].length; j++) {
            let prev = index[words[i - 1][j].charCodeAt() - 'a'.charCodeAt()];
            let curr = index[words[i][j].charCodeAt() - 'a'.charCodeAt()];
            if (prev < curr) {
                valid = true;
                break;
            } else if (prev > curr) {
                return false;
            }
        }
        if (!valid) {
            /* 比较两个字符串的长度 */
            if (words[i - 1].length > words[i].length) {
                return false;
            }
        }
    }
    return true;
};
```
