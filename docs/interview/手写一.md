## 类型判断
```js
typeOf([]) // 'array'
typeOf({}) // 'object'
typeOf(new Date) // 'date'
```
实现:
```js
function typeOf(type){
  let typeStr = Object.prototype.toString.call(type).split(' ')[1]
  return typeStr.slice(0, typeStr.length - 1).toLowerCase()
}
console.log(typeOf([]))
```
## 继承
### 原型继承
```js
function Father(){
  this.name = 'feng'
}
Father.prototype.getName = function () {
  return 'feng1'
}

function Child(){}

Child.prototype = new Father()
let child = new Child()
console.log(child.name)// 缺点:无法向父传参
console.log(child.getName())
```

### 借用构造函数实现继承
```js
function Father(){
  this.name = 'feng'
}
Father.prototype.getName = function () {
  return this.name
}

function Child(){
  Father.call(this)
}

let child = new Child()
console.log(child.name)
console.log(child.getName()) // 缺点不能继承father原型上的属性和方法 每生成一次子实例,父构造函数就调用一次
```

### 寄生继承
```js
function createObj(o){
  let clone = Object.create(o)
  clone.getName = function(){
    console.log('feng')
  }
  return clone
}

let father = {
  name: 'feng'
}

let child = createObj(father)
console.log(child.getName())
```

### 寄生组合继承
```js
function Father(name){
  this.name = name
}
Father.prototype.getItem = function () {
  console.log(this.name)
}
function Child(name){
  Father.call(this, name)
}
Child.prototype = Object.create(Father.prototype)
Child.prototype.constructor = Child

let child = new Child('feng')
console.log(child.name)
```

### Map
```js
function removeCom(arr) {
  let map = new Map(), res = []
  for(let i = 0; i < arr.length; i++) {
    if(!map.has(arr[i])) map.set(arr[i], arr[i])
  }
  map.forEach(item => {
    res.push(item)
  })
  return res
}
```

### filter
```js
function removeCom(arr) {
  let res = arr.filter((cur, index, arr) => {
    return arr.indexOf(cur) === index
  })
  return res
} 
```

### sort
```js
function removeCom(arr) {
  let res = [arr[0]]
  arr.sort()
  for(let i = 1; i < arr.length; i++) {
    if(arr[i] !== arr[i-1]) res.push(arr[i])
  }
  return res
}
```

### includes
```js
function removeCom(arr) {
  let res = []
  for(let i = 0; i < arr.length; i++) {
    if(!res.includes(arr[i])) res.push(arr[i])
  }
  return res
}
```

### 递归
```js
let arr = [1,2,[3,[4]]]
function flatten(arr, res){
  for(let i = 0; i < arr.length; i++) {
    if(Array.isArray(arr[i])){
      flatten(arr[i], res)
    } else {
      res.push(arr[i])
    }
  }
}
let arr1 = []
flatten(arr, arr1)
console.log(arr1)
```
### some
```js
let arr = [1,2,[3,[4]]]
function flatten (arr) {
  while(arr.some(item => Array.isArray())) {
    arr = [].concat(...arr)
  }
  return arr
}
```
### reduce
```js
let arr = [1,2,[3,[4]]]
function flatten(arr) {
  if(!arr.length) return
  return arr.reduce((pre,cur) => 
    Array.isArray(cur)? [...pre, ...flatten(cur)]: [...pre, cur],[]
  )
}
console.log(flatten(arr))
```
## 深浅拷贝
### 浅拷贝

### 深拷贝
```js
function cloneDeep(obj, map = new WeakMap()) {
  if(typeof obj !== 'object' || typeof obj === null || typeof obj === 'function') return obj
  if(typeof obj === 'symbol') return Symbol(obj.description)

  const objFromMap = map.get(obj)
  if(objFromMap) return objFromMap
  let target = {}
  map.set(obj, target)

  if(obj instanceof Map) {
    target = new Map()
    obj.forEach((v, k) => {
      const v1 = cloneDeep(v, map)
      const k1 = cloneDeep(k, map)
      target.set(k1, v1)
    })
  }

  if(obj instanceof set) {
    target = new Set()
    obj.forEach(v => {
      const v1 = cloneDeep(v, map)
      target.add(v1)
    })
  }

  if(obj instanceof Array) {
    target = obj.map(item => cloneDeep(item, map))
  }

  if(obj instanceof Object) {
    for(const key in obj) {
      target[key] = cloneDeep(obj[key], map)
    }
  }

  const symbolKeys = Object.getOwnPrototypeSymbols(obj)
  for(const sKey of symbolKeys) {
    target[sKey] = deepClone(obj[sKey], map)
  }
  return target
}
```
