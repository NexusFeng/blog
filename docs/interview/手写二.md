```js
class EventEmitter {
  constructor() {
    this.events = {};
  }

  // 监听事件
  on(type, cb) {
    if (!this.events) tis.events = Object.create(null);
    if (!this.events[type]) {
      this.events[type] = [cb];
    } else {
      this.events[type].push(cb);
    }
  }
  // 移除事件监听
  off(type, cb) {
    if (!this.events[type]) return;
    this.events[type] = this.events[type].filter((item) => {
      item !== cb;
    });
  }

  // 触发事件监听
  emit(type, ...args) {
    this.events[type] &&
      this.events[type].forEach((fn) => fn.apply(this, args));
  }

  // 只监听一次
  once(type, cb) {
    function fn() {
      cb();
      this.off(type, fn);
    }
    this.on(type, fn);
  }
}
```

### 解析 query 参数

```js
function parseQueryParams(url) {
  let res = {};
  let query = url.substring(url.indexOf('?') + 1).split('&');
  query.forEach((item) => {
    res[item.substring(0, item.indexOf('='))] = item.substring(
      item.indexOf('=') + 1,
    );
  });
  return res;
}

parseQueryParams('http://www.baidu.com?q=foo&bar=baz'); // { q: 'foo', bar: 'baz' }
```

```js
export function parseQueryParams(url) {
  return url
    .split('?')[1]
    .split('&')
    .reduce((target, cur) => {
      const [key, value] = cur.split('=');
      target[key] = value;
      return target;
    }, {});
}
```

```js
let imgs = document.querySelectorAll('img');
function lazyLoad() {
  let container = document.getElementById('container');
  let scrollTop = container.scrollTop;
  let viewHeight = container.innerHeight;
  for (let i = 0; i < imgs.length; i++) {
    if (imgs[i].offsetTop < scrollTop + viewHeight) {
      imgs[i].src = imgs[i].getAttribute('data-src');
      if (i === imgs.length - 1) window.removeEventListener('scroll', lazyLoad);
    }
  }
}
window.addEventListener('scroll', lazyLoad);
```

### 输出最长叠词数量

```js
function getRepititions(str) {
  if (!str) return 0;
  let num = 0,
    set = new Set(),
    start = 0,
    end = 0;
  while (end < str.length) {
    str[end] === str[start] ? set.add(str[end]) : (start = end);
    end++;
    if (start + 1 == end && str[end] !== str[start]) {
      num = Math.max(set.size, num);
      set.clear();
    }
  }
  return num;
}
```
