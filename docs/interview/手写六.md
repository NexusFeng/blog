## 2.实现下面 `run` 和 `ajax` 两个函数

> 考察点：async/await 原理；简易 `co` 模块实现；ajax

```js
/**
 * Ajax简易封装GET请求
 * @param {string} url 请求地址
 * @param {function} cb 回调函数
 * @return {promise} 返回 promise
 */
function ajax(url, cb) {
  const p = new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest()
    xhr.open('GET', url, true)
    xhr.onreadystatechange = function(){
      if(xhr.readyState === 4) {
        resolve(cb())
      } else if(xhr.status === 404) {
        reject(new Error('404'))
      }
    }
    xhr.send(null)
  })
  return p
}
```

```js
function run(gen) {
  gen = gen()
  //可传入多个参数
  // var args = Array.prototype.slice.call(arguments, 1)
  // gen = gen.bind(args)()
  return new Promise((resolve,reject) => {
    onFulfilled()

    function onFulfilled(res) {
      var ret = gen.next(res)
      next(ret)
    }

    function next(res) {
      if (res.done) return resolve(res.value)
      // 自动执行
      Promise.resolve(res.value).then( e => {
        onFulfilled(e)
      } )
    }

  })
}
```

```js
function limitLoad(urls, handle, limit) {
  let total = urls.length;
  let finish = 0,
    requests = [];
  const request = async () => {
    for (let i = 0; i < total; i++) {
      let task = new Promise((resolve) => {
        handle(urls[i]).then((res) => {
          resolve();
        });
      })
        .then(() => {
          let index = requests.findIndex((t) => t === task);
          requests.splice(index, 1);
          finish++;
        })
        .catch(() => {
          let index = requests.findIndex((t) => t === task);
          requests.splice(index, 1);
          requests.push(task);
        });

      requests.push(task);
      if (requests.length === limit) {
        await Promise.race(requests);
      }
    }
  };
  request();
  return new Promise((resolve) => {
    let timer = setInterval(() => {
      if (total == finish) {
        clearInterval(timer);
        resolve("完成");
      }
    }, 500);
  });
}
```

## 5.算法(easy)

### 5.1 两数之和

[传送门](https://leetcode.cn/problems/two-sum/)
```js
var twoSum = function(nums, target) {
  let  map = new Map()
  for(let i = 0; i < nums.length; i++) {
    let num = target - nums[i]
    if(map.has(num)) {
      return [i, map.get(num)]
    }
    map.set(nums[i], i)
  }
  return []
};
```

### 5.2 有效的括号

[传送门](https://leetcode.cn/problems/valid-parentheses/)
```js
var isValid = function(s) {
  if(s.length <=1) return false
  let map = new Map()
  map.set(')', '(')
  map.set(']', '[')
  map.set('}', '{')
  let stack = []
  for(let i = 0; i < s.length; i++) {
    if(map.has(s[i])) {
      let sys = stack.pop()
      if(sys !== map.get(s[i])) return false
    } else {
      stack.push(s[i])
    }
  }
  return !stack.length
}
```
